1. 存储结构的选择 顺序存储：核心在于利用数组的连续性实现高效访问。在栈中，通过 top 指针实现 O(1) 进出；在队列中，为解决循环队列的队满问题，利用取模运算 (rear + 1) % size 有效利用存储空间。 链式存储：核心在于动态内存分配的灵活性。栈采用头插法简化操作；队列通过维护头尾双指针，统一了入队与出队的逻辑，避免了顺序存储带来的空间限制。 2. 逻辑特性与变体 栈的应用（LIFO）：重点在于处理具有“最近相关性”或“逆序性”的问题，比如说下一个更大的元素（NGE问题）。 队列的应用（FIFO）：重点在于处理“顺序性”与“公平调度”问题。 三、实验总结 栈和队列本质上是逻辑受限的线性表，其实现场景（顺序或链式）应根据需求权衡。顺序结构适合规模明确的快速计算，链式结构适合动态变化的复杂业务。 指针与内存管理：在链式栈和队列的 Pop 或 DeQueue 操作中，必须先暂存节点地址再移动指针，最后释放内存，否则会导致内存泄漏。 边界条件：注意判定栈空（top == -1）、栈满、队空（front == rear）及循环队列的队满状态。 表达式计算：需额外注意操作数栈在运算时的弹出顺序。

1. 顺序栈

  • 思路：创建一个支持初始化、入、出、取顶、销毁的顺序存储栈。

  • 要点：分配连续数组空间，维护 top 指针（初始为 -1）。入栈前需检查是否溢出，出栈则移动指针。 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef int Elemtype;

typedef struct {
    Elemtype* base;
    int size;
    int top;
} SqStack;

void Push( SqStack *s, Elemtype x )
{
    s->top ++;
    s->base[ s->top ] = x;
}

int Pop( SqStack *s, Elemtype *x )
{
    if( s->top == -1 ) return 0;
    *x = s->base[s->top];
    s->top --;
    return 1;
}

int GetTop( SqStack s, Elemtype *x )
{
    if( s.top == -1 ) return 0;
    *x = s.base[s.top];
    return 1;
}

3. 顺序队列

  • 思路：实现一个基础的顺序存储队列。

  • 要点：维护 front 和 rear 指针。入队在队尾 rear 处，出队在队头 front 处。 

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef int Elemtype;

typedef struct SqQueue {
    Elemtype* data;
    int size;
    int front, rear;
} SqQueue;

int Empty( SqQueue *q )
{
    return q->front == q->rear;
}

void EnQueue( SqQueue* q, Elemtype x )
{
    q->data[ q->rear ] = x;
    q->rear ++;
}

int Front( SqQueue* q )
{
    if( Empty(q) ) return -1;
    return q->data[ q->front ];
}

int Back( SqQueue* q )
{
    if( Empty(q) ) return -1;
    return q->data[ q->rear-1 ];
}

int DeQueue( SqQueue* q )
{
    int x = q->data[ q->front ];
    q->front ++;
    return x;
}

void InitQueue( SqQueue* q )
{
    q->size = 500;
    q->data = ( Elemtype* )malloc( sizeof( Elemtype ) * (q->size) );
    q->front = 0;
    q->rear = 0;

4. 链式队列

  • 思路：实现一个带头结点的链式队列，支持大容量数据。

  • 要点：front 指向头结点，tail 指向末尾。入队操作挂在 tail->next；出队从 front->next 取值。 

#include<stdlib.h>
#include<stdio.h>
#include<string.h>

typedef int Elemtype;

typedef struct QNode {
    Elemtype data;
    struct QNode *next;
} QNode;

typedef struct {
    QNode* front;
    QNode* tail;
} LinkQueue;

int Empty( LinkQueue &q )
{
    return q.front == q.tail;
}

void EnQueue( LinkQueue &q, Elemtype x )
{
    QNode* p = (QNode*)malloc(sizeof(QNode));
    p->data = x;
    p->next = NULL;
    q.tail->next = p;
    q.tail = p;
}

int DeQueue( LinkQueue &q )
{
    QNode* p = q.front->next;
    int x = p->data;
    q.front->next = p->next;
    if( q.tail == p ) q.tail = q.front;
    free( p );
    return x;
}

int Front( LinkQueue &q )
{
    return q.front->next->data;
}

int Back( LinkQueue &q )
{
    return q.tail->data;
}

5. 循环队列

  • 思路：通过模运算处理循环操作。

  • 要点：判断队满条件为 (rear + 1) % size == front，计算长度使用 (rear - front + size) % size。 

#include<stdio.h>
#include<stdlib.h>
#include<string.h>

typedef int Elemtype;

typedef struct SqQueue {
    Elemtype* base;
    int front, rear;
    int size;
} SqQueue;

void InitQueue( SqQueue *q )
{
    q->size = 5;
    q->base = ( Elemtype* )malloc( sizeof( Elemtype ) * (q->size) );
    q->front = 0;
    q->rear = 0;
}

int EnQueue( SqQueue *q, Elemtype x )
{
    if( ( q->rear + 1 ) % q->size == q->front ) return 0;
    q->base[ q->rear ] = x;
    q->rear = ( q->rear + 1 ) % q->size;
    return 1;
}

int DeQueue( SqQueue *q, Elemtype *x )
{
    if( q->front == q->rear ) return 0;
    *x = q->base[ q->front ];
    q->front = ( q->front + 1 ) % q->size;
    return 1;
}

int Destroy( SqQueue *q )
{
    free(q->base);
}

int QueueLength( SqQueue q )
{
    return ( q.rear - q.front + q.size ) % q.size;
}

8. 表达式转换（中-->后）

  • 思路：将中缀式转换为后缀式。

  • 要点：利用运算符栈。根据优先级决定弹栈时机，括号处理需强制改变优先级边界。 

#include <bits/stdc++.h>
using namespace std;
using ll = long long;

const int maxn = 2e5 + 5;

int get( char op )
{
    if( op == '+' || op == '-' ) return 1;
    if( op == '*' || op == '/' ) return 2;
    return 0;
}

void solve( )
{
    string s;
    while( getline( cin, s ) )
    {
        if( s.empty( ) ) continue;
        vector< string > res;
        stack< char > ops;

        for( int i = 0; i < s.size( ); ++ i )
        {
            char c = s[i];
            if( isdigit( c ) ) {
                res.push_back( string( 1, c ) );
            } else if( c == '(' ) {
                ops.push( c );
            } else if( c == ')' ) {
                while( !ops.empty( ) && ops.top( ) != '(' ) {
                    res.push_back( string( 1, ops.top( ) ) );
                    ops.pop( );
                }
                if( !ops.empty( ) ) ops.pop( );
            } else {
                while( !ops.empty( ) && get( ops.top( ) ) >= get( c ) ) {
                    res.push_back( string( 1, ops.top( ) ) );
                    ops.pop( );
                }
                ops.push( c );
            }
        }
        while( !ops.empty( ) ) {
            res.push_back( string( 1, ops.top( ) ) );
            ops.pop( );
        }
        for( int i = 0; i < res.size( ); ++ i )
        {
            if( i ) cout << " ";
            cout << res[i];
        }
        cout << '\n';
    }
}

int main( )
{
    ios::sync_with_stdio( 0 ); cin.tie( 0 );
    solve( );
    return 0;
}

9. Hero In Maze

  • 思路：在给定迷宫地图中寻找起点到终点的可行路径。

  • 要点：使用递归实现 DFS（深度优先搜索），利用系统栈记录回溯过程，配合访问标记数组。 

#include <bits/stdc++.h>
using namespace std;

const int maxn = 20 + 5;

int n, m, t;
char maze[maxn][maxn];
bool vis[maxn][maxn];
bool found;

int dx[] = {0, 0, 1, -1};
int dy[] = {1, -1, 0, 0};

bool check( int x, int y )
{
    return 1 <= x && x <= m && 1 <= y && y <= n;
}

void dfs( int x, int y, int step )
{
    // cout << x << ' ' << y << ' ' << step << '\n';
    if( found || step > t ) return ;

    if( maze[x][y] == 'P' ) { found = true; return; }

    for( int d = 0; d < 4; ++ d )
    {
        int nx = x + dx[d], ny = y + dy[d];

        if( check( nx, ny ) && !vis[nx][ny] && maze[nx][ny] != '*' )
        {
            vis[nx][ny] = 1;
            dfs(nx, ny, step + 1);
            vis[nx][ny] = 0;
        }
    }
}

void solve( )
{
    if( !n && !m && !t ) return;

    int stx = 1, sty = 1;
    found = false;
   
    for( int i = 1; i <= m; ++ i )
    {
        for( int j = 1; j <= n; ++ j )
        {
            vis[i][j] = 0;
            cin >> maze[i][j];
            if( maze[i][j] == 'S' ) stx = i, sty = j;
        }
    }

    vis[stx][sty] = true;
    dfs( stx, sty, 0 );

    if( found ) cout << "YES" << '\n';
    else cout << "NO" << '\n';
}

int main() {
    ios::sync_with_stdio(0); cin.tie(0);
    while( cin >> n >> m >> t ) solve( );
    return 0;
}

11. 简单计算器

  • 思路：直接求解中缀表达式的值，无需显式转换为后缀。

  • 要点：使用操作数栈和运算符栈。遇到低优先级符号时，先结算栈内高优先级运算。 

#include <bits/stdc++.h>
using namespace std;
using ll = long long;

const int maxn = 2e5 + 5;

int get( string opt )
{
    if( opt == "+" || opt == "-" ) return 1;
    if( opt == "*" || opt == "/" ) return 2;
    return 0;
}

double calc( double a, double b, string opt )
{
    if( opt == "+" ) return a + b;
    if( opt == "-" ) return a - b;
    if( opt == "*" ) return a * b;
    if( opt == "/" ) return a / b;
    return 0;
}

void solve( )
{
    string str;
    while( getline( cin, str ) )
    {
        if( str.length( ) == 1 && str[0] == '0' ) break;
        stringstream ss( str );
        string cur;
        stack< double > nums;
        stack< string > ops;
        while( ss >> cur )
        {
            if( isdigit( cur[0] ) ) nums.push( stod( cur ) );
            else {
                while( !ops.empty( ) && get( ops.top( ) ) >= get( cur ) )
                {
                    double b = nums.top( ); nums.pop( );
                    double a = nums.top( ); nums.pop( );
                    nums.push( calc( a, b, ops.top( ) ) );
                    ops.pop( );
                }
                ops.push( cur );
            }
        }
        while( !ops.empty( ) )
        {
            double b = nums.top( ); nums.pop( );
            double a = nums.top( ); nums.pop( );
            nums.push( calc( a, b, ops.top( ) ) );
            ops.pop( );
        }
        cout << fixed << setprecision( 2 ) << nums.top( ) << '\n';
    }
}

int main( )
{
    ios::sync_with_stdio( 0 ); cin.tie( 0 );
    solve( );
    return 0;
